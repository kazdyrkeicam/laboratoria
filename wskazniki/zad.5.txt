void printArr(int *begin, int *end){

		printf("%d ", i);
}

void swap(int *a, int *b){
	*b ^= *a;
	*a ^= *b;
	*b ^= *a;
}

int *f(int arr[], int n, bool *changed){
	int *even;
	*changed = false;

	do{
		for(int i = 0; i < n; ++i){
			if(*(arr + i) % 2 == 0 && *(arr + i + 1) % 2 == 1){
				swap(arr + i, arr + i + 1);
				*changed = true;
				even = arr + i + 1;
			}
		}
		n--;
	}while(n > 0);
	return even;
}
